/* 01 - TOGGLE DROPDOWN & FETCH EMPLOYEES */ window.toggleCoverDropdown = async function() { const dd = document.getElementById("coverDropdown"); const select = document.getElementById("assignmentSelect"); if (dd.style.display === "none") { dd.style.display = "block"; // Fetch employee names from Firebase to fill the options try { const q = query(collection(db, "employees")); const snap = await getDocs(q); // Clear old options and add the default select.innerHTML = ''; snap.forEach(doc => { const emp = doc.data(); // Only show people who have an assignment set up if (emp.assignment) { const opt = document.createElement("option"); opt.value = emp.assignment; opt.innerText = `${emp.firstName} (${emp.assignment})`; select.appendChild(opt); } }); } catch (e) { console.error("Error loading employees:", e); } } else { dd.style.display = "none"; } } /* 02 - CHANGE ASSIGNMENT (TRIGGER) */ window.changeAssignment = function(newVal) { if (!newVal) return; // Update the session locally so V2 and V3 refresh with the new data const user = JSON.parse(sessionStorage.getItem("currentEmployee")); user.assignment = newVal; sessionStorage.setItem("currentEmployee", JSON.stringify(user)); // Close dropdown and refresh the view document.getElementById("coverDropdown").style.display = "none"; location.reload(); }
